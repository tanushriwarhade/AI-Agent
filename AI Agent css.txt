import React, { useState, useEffect, useRef } from 'react';
import {
  Calendar,
  Mic,
  Activity,
  CheckCircle,
  Clock,
  ShieldAlert,
  Music,
  Volume2,
  BrainCircuit,
  Zap,
  MicOff,
} from 'lucide-react';

const App = () => {
  // --- STATE MANAGEMENT ---
  const [energyLevel, setEnergyLevel] = useState(85); // 0-100
  const [mood, setMood] = useState('energetic'); // 'energetic' | 'drained'
  const [isListening, setIsListening] = useState(false);
  const [agentThought, setAgentThought] = useState(
    'System initialized. Monitoring user bio-rhythm...'
  );
  const [showGatekeeperModal, setShowGatekeeperModal] = useState(false);

  // --- TEXT TO SPEECH ENGINE ---
  const speak = (text) => {
    if ('speechSynthesis' in window) {
      window.speechSynthesis.cancel(); // Stop previous speech
      const utterance = new SpeechSynthesisUtterance(text);
      // Try to get a Google voice, otherwise default
      const voices = window.speechSynthesis.getVoices();
      const preferred =
        voices.find((v) => v.name.includes('Google US English')) || voices[0];
      utterance.voice = preferred;
      utterance.pitch = 0.9; // Slight robotic depth
      utterance.rate = 1.0;
      window.speechSynthesis.speak(utterance);
    }
  };

  // --- VOICE LOGIC (Hybrid: Real + Backup) ---
  const handleVoiceInteraction = () => {
    // 1. Try Real Speech Recognition
    const SpeechRecognition =
      window.SpeechRecognition || window.webkitSpeechRecognition;

    if (!SpeechRecognition) {
      // Fallback for browsers without support
      runSimulation();
      return;
    }

    const recognition = new SpeechRecognition();
    recognition.lang = 'en-US';
    recognition.interimResults = false;
    recognition.maxAlternatives = 1;

    setIsListening(true);
    setAgentThought('Microphone active. Listening for mood cues...');

    recognition.start();

    recognition.onresult = (event) => {
      const transcript = event.results[0][0].transcript.toLowerCase();
      analyzeMood(transcript);
      setIsListening(false);
    };

    recognition.onerror = (event) => {
      console.error('Mic Error:', event.error);
      setIsListening(false);
      // IF BLOCKED ('not-allowed'), RUN SIMULATION AUTOMATICALLY
      if (
        event.error === 'not-allowed' ||
        event.error === 'service-not-allowed'
      ) {
        setAgentThought('Microphone blocked. Engaging Simulation Protocol...');
        runSimulation();
      } else {
        setAgentThought(`Voice Error: ${event.error}. Try again.`);
      }
    };

    recognition.onend = () => {
      setIsListening(false);
    };
  };

  // --- THE LOGIC BRAIN ---
  const analyzeMood = (text) => {
    const negativeWords = [
      'tired',
      'stress',
      'burnout',
      'no',
      'hard',
      'drained',
      'bad',
    ];
    const positiveWords = [
      'ready',
      'go',
      'energy',
      'good',
      'excited',
      'yes',
      'fast',
    ];

    const isNegative = negativeWords.some((w) => text.includes(w));
    const isPositive = positiveWords.some((w) => text.includes(w));

    if (isNegative) {
      triggerDrainedState(text);
    } else if (isPositive) {
      triggerEnergeticState(text);
    } else {
      setAgentThought(
        `Input: "${text}" -> Analysis: Neutral. Asking for clarification.`
      );
      speak(
        "I didn't catch a specific mood. Are you ready to work, or do you need rest?"
      );
    }
  };

  // --- SIMULATION MODE (For Foolproof Demo) ---
  const runSimulation = () => {
    // This runs if the Mic fails or is blocked
    setIsListening(true);
    setAgentThought('Simulating Voice Input (Demo Mode)...');

    setTimeout(() => {
      setIsListening(false);
      // Toggle state based on current mood
      if (mood === 'energetic') {
        triggerDrainedState('I am feeling really tired and stressed today.');
      } else {
        triggerEnergeticState('I am ready to crush this!');
      }
    }, 2000);
  };

  // --- STATE TRIGGERS ---
  const triggerDrainedState = (input) => {
    setMood('drained');
    setEnergyLevel(20);
    setAgentThought(
      `Input: "${input}" -> Detected: High Stress -> ACTION: ACTIVATING GATEKEEPER.`
    );
    speak(
      'I hear high stress levels. I am locking your calendar to protect your health.'
    );
  };

  const triggerEnergeticState = (input) => {
    setMood('energetic');
    setEnergyLevel(95);
    setAgentThought(
      `Input: "${input}" -> Detected: High Energy -> ACTION: HIGH PERFORMANCE MODE.`
    );
    speak('Energy confirmed. Optimizing interface for deep work.');
  };

  // --- GATEKEEPER ---
  const handleDateClick = () => {
    if (mood === 'drained') {
      setShowGatekeeperModal(true);
      setAgentThought(
        'INTERVENTION: Blocked user action due to Critical Energy Levels.'
      );
      speak('I cannot allow that. Your energy is too low. Please rest.');
    } else {
      setAgentThought('Opening schedule for new task entry...');
    }
  };

  // --- STYLES ---
  const accentColor = mood === 'energetic' ? 'text-cyan-400' : 'text-amber-500';
  const glowColor =
    mood === 'energetic' ? 'shadow-cyan-500/50' : 'shadow-amber-500/50';
  const orbColor = mood === 'energetic' ? 'bg-cyan-500' : 'bg-amber-600';
  const borderColor =
    mood === 'energetic' ? 'border-cyan-500/30' : 'border-amber-500/30';

  return (
    <div
      className={`min-h-screen bg-slate-950 text-white font-sans selection:bg-cyan-500/30 flex overflow-hidden transition-colors duration-1000 ease-in-out`}
    >
      {/* --- LEFT SIDEBAR --- */}
      <aside className="w-1/5 border-r border-white/10 bg-white/5 backdrop-blur-xl flex flex-col p-6 gap-8 relative z-10">
        <div className="flex items-center gap-3">
          <div
            className={`w-8 h-8 rounded-lg bg-gradient-to-br ${
              mood === 'energetic'
                ? 'from-cyan-400 to-blue-600'
                : 'from-amber-500 to-red-600'
            } transition-all duration-1000`}
          ></div>
          <h1 className="font-bold text-xl tracking-tight">FlowState</h1>
        </div>

        {/* LIQUID ORB */}
        <div className="flex flex-col items-center justify-center flex-1">
          <div className="relative w-48 h-48 flex items-center justify-center">
            {/* Background Glow */}
            <div
              className={`absolute w-full h-full rounded-full opacity-40 blur-3xl transition-all duration-1000 ${orbColor} animate-pulse`}
            ></div>

            {/* The Orb Itself */}
            <div
              className={`relative w-32 h-32 ${orbColor} transition-all duration-1000 rounded-full flex items-center justify-center shadow-[0_0_40px_rgba(0,0,0,0.5)]`}
              style={{
                animation: isListening
                  ? 'pulse 0.5s infinite alternate'
                  : mood === 'energetic'
                  ? 'morph 4s infinite ease-in-out'
                  : 'morph 10s infinite ease-in-out',
                borderRadius: '60% 40% 30% 70% / 60% 30% 70% 40%',
              }}
            >
              <Zap
                className={`w-12 h-12 text-white transition-opacity duration-500 ${
                  mood === 'energetic' ? 'opacity-100' : 'opacity-50'
                }`}
              />
            </div>

            {/* Battery Ring */}
            <svg className="absolute w-48 h-48 -rotate-90">
              <circle
                cx="96"
                cy="96"
                r="88"
                stroke="currentColor"
                strokeWidth="2"
                className="text-white/10"
                fill="none"
              />
              <circle
                cx="96"
                cy="96"
                r="88"
                stroke="currentColor"
                strokeWidth="4"
                className={`${accentColor} transition-all duration-1000`}
                strokeDasharray="552"
                strokeDashoffset={552 - (552 * energyLevel) / 100}
                strokeLinecap="round"
                fill="none"
              />
            </svg>
          </div>

          <div className="mt-6 text-center">
            <h2
              className={`text-3xl font-bold transition-colors duration-500 ${accentColor}`}
            >
              {energyLevel}%
            </h2>
            <p className="text-sm text-slate-400 uppercase tracking-widest text-xs mt-1">
              Bio-Battery
            </p>
          </div>
        </div>

        {/* Voice Control */}
        <div className="space-y-4">
          <button
            onClick={handleVoiceInteraction}
            className={`w-full py-4 rounded-xl flex items-center justify-center gap-3 font-semibold transition-all duration-300 shadow-lg ${
              isListening
                ? 'bg-red-500 scale-95 shadow-red-500/20'
                : 'bg-white/10 hover:bg-white/20 hover:scale-105'
            }`}
          >
            {isListening ? <MicOff className="animate-pulse" /> : <Mic />}
            {isListening ? 'Listening...' : 'Check Vibe'}
          </button>

          {/* Ambient Player */}
          <div
            className={`p-4 rounded-xl border border-white/5 bg-black/40 flex items-center gap-3 transition-all duration-1000 ${
              mood === 'drained' ? 'opacity-100 translate-y-0' : 'opacity-60'
            }`}
          >
            <Music size={18} className={accentColor} />
            <div className="flex-1 overflow-hidden">
              <p className="text-[10px] text-slate-400 uppercase tracking-wider">
                Now Playing
              </p>
              <p className="text-sm font-medium whitespace-nowrap overflow-hidden text-ellipsis">
                {mood === 'energetic'
                  ? '‚ö° Cyberpunk Focus Mix'
                  : 'üåßÔ∏è Heavy Rain & Theta Waves'}
              </p>
            </div>
            <div className="flex gap-0.5 items-end h-4">
              {[1, 2, 3].map((i) => (
                <div
                  key={i}
                  className={`w-1 ${orbColor} animate-pulse`}
                  style={{
                    height: `${Math.random() * 100}%`,
                    animationDelay: `${i * 0.1}s`,
                  }}
                ></div>
              ))}
            </div>
          </div>
        </div>
      </aside>

      {/* --- CENTER PANEL --- */}
      <main className="flex-1 p-8 flex flex-col gap-6 relative overflow-y-auto bg-gradient-to-b from-slate-900 to-slate-950">
        <header className="flex justify-between items-end pb-6 border-b border-white/5">
          <div>
            <h2 className="text-3xl font-bold tracking-tight">
              Weekly Schedule
            </h2>
            <p className="text-slate-400 mt-1">October 24 - 30, 2025</p>
          </div>
          <div
            className={`px-4 py-2 rounded-full border ${borderColor} ${accentColor} bg-white/5 text-sm font-medium flex items-center gap-2 transition-all duration-500`}
          >
            {mood === 'energetic' ? (
              <Zap size={16} />
            ) : (
              <ShieldAlert size={16} />
            )}
            Mode:{' '}
            {mood === 'energetic' ? 'High Performance' : 'Burnout Protection'}
          </div>
        </header>

        {/* Calendar Grid */}
        <div
          className={`flex-1 grid grid-cols-1 gap-4 transition-all duration-1000 ${
            mood === 'drained' ? 'grayscale opacity-70 blur-[1px]' : ''
          }`}
        >
          {['09:00 AM', '11:00 AM', '02:00 PM', '04:00 PM', '05:00 PM'].map(
            (time, i) => (
              <div
                key={i}
                onClick={handleDateClick}
                className="group p-5 rounded-xl bg-white/5 border border-white/5 hover:border-white/10 hover:bg-white/10 cursor-pointer transition-all flex gap-6 items-center"
              >
                <span className="text-sm text-slate-500 font-mono w-20">
                  {time}
                </span>
                <div className="h-full w-0.5 bg-white/10 group-hover:bg-cyan-500/50 transition-colors"></div>

                {i === 1 ? (
                  <div
                    className={`flex-1 p-3 rounded-lg ${
                      mood === 'energetic'
                        ? 'bg-cyan-500/10 border border-cyan-500/30'
                        : 'bg-slate-800 border border-slate-700'
                    }`}
                  >
                    <h4 className="font-semibold text-white">
                      Product Strategy Sync
                    </h4>
                    {mood === 'drained' && (
                      <span className="text-xs text-amber-400 flex items-center gap-1 mt-1">
                        <ShieldAlert size={10} /> Suggest Rescheduling
                      </span>
                    )}
                  </div>
                ) : (
                  <div className="text-slate-600 text-sm italic group-hover:text-slate-400 transition-colors">
                    {mood === 'drained'
                      ? 'Slot Locked by Agent'
                      : 'Available Slot'}
                  </div>
                )}
              </div>
            )
          )}
        </div>

        {/* GATEKEEPER MODAL */}
        {showGatekeeperModal && (
          <div className="absolute inset-0 bg-slate-950/90 backdrop-blur-md z-50 flex items-center justify-center p-8 animate-in fade-in zoom-in duration-300">
            <div className="bg-slate-900 border border-amber-500/50 p-8 rounded-2xl max-w-md shadow-2xl shadow-amber-900/40 text-center ring-1 ring-amber-500/20">
              <ShieldAlert className="w-20 h-20 text-amber-500 mx-auto mb-6 animate-pulse" />
              <h3 className="text-3xl font-bold text-white mb-2">
                Access Denied
              </h3>
              <p className="text-amber-500 font-mono text-sm mb-6 uppercase tracking-widest">
                Energy Critical (20%)
              </p>

              <p className="text-slate-400 mb-8 leading-relaxed">
                My sensors detect high vocal stress. I cannot allow you to book
                new meetings.
                <br />
                <span className="text-slate-500 text-sm mt-2 block">
                  Rest is required for optimal future output.
                </span>
              </p>

              <div className="bg-black/30 p-4 rounded-lg text-left mb-8 border border-white/5">
                <p className="text-[10px] text-slate-500 mb-2 uppercase font-bold">
                  Auto-Drafted Reply:
                </p>
                <p className="text-sm italic text-amber-200">
                  "I'm currently at capacity. Let's look at next week when I can
                  give this my full attention."
                </p>
              </div>

              <button
                onClick={() => setShowGatekeeperModal(false)}
                className="w-full py-3 bg-amber-600 hover:bg-amber-500 rounded-xl font-bold text-white transition-all shadow-lg shadow-amber-900/50"
              >
                Acknowledge & Close
              </button>
            </div>
          </div>
        )}
      </main>

      {/* --- RIGHT SIDEBAR --- */}
      <aside className="w-[28%] bg-black/20 backdrop-blur-xl border-l border-white/5 p-6 flex flex-col gap-6">
        {/* Agent Thought Stream */}
        <div className="bg-slate-900/80 p-5 rounded-xl border border-white/10 min-h-[180px] shadow-inner relative overflow-hidden">
          <div className="absolute top-0 left-0 w-full h-1 bg-gradient-to-r from-transparent via-cyan-500 to-transparent opacity-50"></div>
          <div className="flex items-center gap-2 mb-4 text-xs font-bold tracking-wider text-slate-500 uppercase">
            <BrainCircuit
              size={14}
              className={
                mood === 'energetic' ? 'text-cyan-400' : 'text-amber-500'
              }
            />
            Neural Process Log
          </div>
          <p className="font-mono text-sm leading-relaxed text-cyan-100/90 typing-effect">
            <span className="text-slate-500 mr-2">
              {new Date().toLocaleTimeString([], {
                hour: '2-digit',
                minute: '2-digit',
              })}{' '}
              {'>'}
            </span>
            {agentThought}
          </p>
        </div>

        {/* Task Hub */}
        <div className="flex-1">
          <h3 className="font-bold mb-4 flex items-center gap-2 text-sm uppercase tracking-wider text-slate-400">
            <CheckCircle size={16} /> Pending Optimization
          </h3>
          <div className="space-y-3">
            {[1, 2, 3].map((task) => (
              <div
                key={task}
                className="flex items-center gap-3 p-3 rounded-xl bg-white/5 hover:bg-white/10 transition-colors group border border-transparent hover:border-white/5"
              >
                <div
                  className={`w-4 h-4 rounded-full border-2 ${borderColor} group-hover:bg-white/10 transition-colors`}
                ></div>
                <div className="flex-1">
                  <div className="h-2 w-3/4 bg-slate-800 rounded mb-2 group-hover:w-full transition-all duration-500"></div>
                  <div className="h-1.5 w-1/3 bg-slate-800/50 rounded"></div>
                </div>
              </div>
            ))}
          </div>
        </div>

        {/* Stats */}
        <div className="grid grid-cols-2 gap-4">
          <div className="p-4 bg-white/5 rounded-xl border border-white/5">
            <Activity className="text-slate-400 mb-2" size={20} />
            <div className="text-2xl font-bold">4.2h</div>
            <div className="text-xs text-slate-500">Deep Work</div>
          </div>
          <div className="p-4 bg-white/5 rounded-xl border border-white/5">
            <Clock className="text-slate-400 mb-2" size={20} />
            <div className="text-2xl font-bold">2</div>
            <div className="text-xs text-slate-500">Meetings Left</div>
          </div>
        </div>
      </aside>

      {/* GLOBAL ANIMATION STYLES */}
      <style>{`
        @keyframes morph {
          0% { border-radius: 60% 40% 30% 70% / 60% 30% 70% 40%; }
          50% { border-radius: 30% 60% 70% 40% / 50% 60% 30% 60%; }
          100% { border-radius: 60% 40% 30% 70% / 60% 30% 70% 40%; }
        }
        .typing-effect {
          display: inline-block;
          overflow: hidden;
          white-space: pre-wrap;
        }
      `}</style>
    </div>
  );
};

export default App;
